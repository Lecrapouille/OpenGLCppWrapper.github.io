32  //! \brief Class defining a Scene node.
33  //!
34  //! A scene is a spatial representation of a graphical scene using a tree
35  //! structure. Instead of storing 3D models (entities) in an array, where
36  //! their positions are related to the world origin, making more difficult to
37  //! move these entites inside the world, a scene classifies entities inside a
38  //! tree structure where parent-children relations between nodes allow to
39  //! define relative transformations of their positions given by a transform
40  //! matrix (translation, rotation and scaling). This way, the animation of
41  //! entities are easier because modifying the matrix of one done will
42  //! automatically impact the position of the descendant nodes. Computed
43  //! matrices are usually passed to the GLSL shaders through a 4x4 uniform
44  //! matrix with the name of "model matrix".
45  //!
46  //! For more information you can read this introduction document:
47  //! https://research.ncl.ac.uk/game/mastersdegree/graphicsforgames/scenegraphs/Tutorial%206%20-%20Scene%20Graphs.pdf
48  //! The code source of this file largely get inspired by this document.
49  // ***************************************************************************

165  //--------------------------------------------------------------------------
166  //! \brief Traverse recursively the scene for computing all world
167  //! transformation matrices for traversable nodes. Non traversable nodes and
168  //! their descendant are ignored.
169  //!
170  //! \note no traversable nodes are ignored as well as their descendants.
171  //!
172  //! \param dt: delta time from the previous update.
173  //--------------------------------------------------------------------------
174  void update(float const dt)
175  {
176  if (unlikely(!isTraversable()))
177  return;
178 
179  // Derived class may override this function for animating nodes.
180  doUpdate(dt);
181 
182  // Update the matrix transform from the parent matrix.
183  m_world_transform = Transformable3D::transform();
184  if (likely(hasParent()))
185  m_world_transform *= parent().m_world_transform;
186 
187  for (auto& i: m_children)
188  i->update(dt);
189  }

191  //--------------------------------------------------------------------------
192  //! \brief Callback triggered when visiting the scene for updating
193  //! transformation matrices. This method does nothing by default but can be
194  //! overrided by Node3D's derived classes for animating/moving the current
195  //! node (which will impact on position of descendant nodes).
196  //!
197  //! \note: the project https://github.com/vahlers/scg3 traverse the tree
198  //! using the vistor design pattern. I did not follow because I am a big fan
199  //! of visitors: in my personal opinion they make the code harder to
200  //! understand. In the course https://research.ncl.ac.uk/game/mastersdegree/
201  //! graphicsforgames/scenegraphs/Tutorial%206%20-%20Scene%20Graphs.pdf
202  //! they let derived classes override the method update() but suppose the
203  //! developper will call the method update() of the parent class. I think this
204  //! is an error prone pratice. I think doUpdate() does the same thing in a more
205  //! simpler way for the developper.
206  //!
207  //! \param dt: delta time from the previous update.
208  //--------------------------------------------------------------------------

273  //--------------------------------------------------------------------------
274  //! \brief Attach the given node as child to this instance and trig the
275  //! callback onNodeAdded().
276  //!
277  //! \note A node cannot be attached to itself or on an already attached
278  //! node. This constrain avoid to create a cycle making the traversal of
279  //! the scene more difficult.
280  //!
281  //! \param node the node which will be the child node. It will placed as the
282  //! last child.
283  //!
284  //! \note Child hierarchy order is important because the tree is traversed
285  //! using a depth-first traversal with pre-order.
286  //!
287  //! \return *this for combining several adds.
288  //--------------------------------------------------------------------------
289  Node3D& attach(Node3D_SP node)
290  {
291  if (likely(this == node.get()))
292  {
293  ERROR("Cannot attach a node to itself");
294  return *this;
295  }
296 
297  if (unlikely(node->m_attached))
298  {
299  ERROR("Node already attached");
300  return *this;
301  }
302 
303  node->m_parent = this; // TODO: shared_from_this();
304  m_children.push_back(node);
305  onNodeAdded(node);
306  return *this;
307  }

310  //! \brief Remove a node and its descendants from the scene. Memory is
311  //! automatically released if nobody is refering the deleted node.
312  //--------------------------------------------------------------------------
313  Node3D& detach(Node3D_SP node)
314  {

322  return *this;

325  //--------------------------------------------------------------------------
326  //! \brief Execute the function on the node and each of its descendants.
327  //! \note depth-first traversal with pre-order.
328  //! \param node the node to start with.
329  //! \param functor the function to call for each traversed node.
330  //--------------------------------------------------------------------------
331  template<typename Functor>
332  static void traverse(Node3D_SP node, Functor functor)
333  {
334  if (unlikely(!node->isTraversable()))
335  return;
336 
337  functor(node);
338  for (auto& c: node->children())
339  traverse(c, functor);
340  }

342  //--------------------------------------------------------------------------
343  //! \brief Execute the function on ancestors of the given node.
344  //! \param node the node to start with.
345  //! \note the function is not called on the node passed as parameter.
346  //! \param functor the function to call for each traversed node.
347  //--------------------------------------------------------------------------
348  template<typename Functor>
349  static void traverseAncestors(Node3D_SP node, Functor functor)
350  {
351  Node3D* n = node.get();
352  while (likely(n->hasParent()))
353  {
354  n = n->m_parent;
355  functor(*n);
356  }
357  }

409  //--------------------------------------------------------------------------
410  //! \brief Find recursively the first node matching the predicate. Halt on
411  //! the first match.
412  //!
413  //! \param root the node to start searching.
414  //! \param predicate the functor taking a scene node as parameter and
415  //! returning a bool whome the value depends on the matching result.
416  //!
417  //! \return the node on success, else return nullptr.
418  //--------------------------------------------------------------------------
419  template<typename Functor>
420  static Node3D_SP findNode(Node3D_SP root, Functor predicate)
421  {
422  if (predicate(root))
423  return root;
424 
425  for (auto& c: root->m_children)
426  {
427  Node3D_SP node = findNode(c, predicate);
428  if (node != nullptr)
429  return node;
430  }
431 
432  return nullptr;
433  }

450  //--------------------------------------------------------------------------
451  //! \brief Return the node following the given child path.
452  //!
453  //! \param root the node to start searching with.
454  //!
455  //! \param path the child path: a set of node names separating by '/'. i.e.
456  //! "foo/bar". Note that traillings '/' are ignored i.e."foo/bar/" or
457  //! "foo/bar/////". TODO If the path starts by '/' so the search is made from the

459  //!

462  static Node3D_SP getNode(Node3D_SP node, std::string const& path)
463  {
464  std::stringstream sspath(path);
465  std::string name;
466  Node3D_SP child = nullptr;
467  Node3D_SP n = node; // FIXME: (path[0] == '/') ? root(node) : node;
468 
469  while (std::getline(sspath, name, '/'))
470  {
471  if (name == "")
472  continue;
473 
474  child = n->getChild(name);
475  if (child == nullptr)
476  return nullptr;
477 
478  n = child;
479  }
480  return child;
481  }
