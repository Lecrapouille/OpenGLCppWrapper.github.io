30 //! \brief a Transformable defines a 4x4 transformation matrix from a translation, a rotation and a scale.

263  //! \brief Return the 4x4 transform matrix combining the
264  //! position/rotation/scale/origin of the object.
265  //!
266  //! Operation:
267  //! Transform = TranslationMatrix * RotationMatrix * ScaleMatrix;
268  //! TransformedVector = Transform * OriginalVector;
269  //! Note:
270  //! Be careful of operation order: we apply scale first, then
271  //! the rotation then the translation.
272  Matrix<T, n + 1_z, n + 1_z> const& transform()
273  {
274  if (unlikely(m_transform_needs_update))
275  {
276  Matrix<T, n+1_z, n+1_z> I(matrix::Identity);
277  m_transform = matrix::translate(I, m_position - m_origin);
278  m_transform = matrix::rotate(m_transform, m_angle, m_axis);
279  m_transform = matrix::scale(m_transform, m_scale);
280  m_transform_needs_update = false;
281  m_inverse_trans_needs_update = true;
282  }
283 
284  return m_transform;
285  }
286 
287  //! \brief Return the 4x4 inverse transform matrix.
288  Matrix<T, n + 1_z, n + 1_z> const& invTransform()
289  {
290  if (unlikely(m_inverse_trans_needs_update))
291  {
292  if (unlikely(!matrix::inverse(transform(), m_inverse_transform)))
293  {
294  ERROR("Failed to inverse the matrix");
295  matrix::identity(m_inverse_transform);
296  }
297  m_inverse_trans_needs_update = false;
298  }
299  return m_inverse_transform;
300  }
