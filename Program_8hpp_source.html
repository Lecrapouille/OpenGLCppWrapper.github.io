165  //----------------------------------------------------------------------------
166  //! \brief Bind VAO with this GLProgram instance.
167  //!
168  //! Two cases:
169  //! - If it's the first time that the VAO is bound to the GLProgram then the
170  //! VAO has its internal list of VBOs and textures populated. They are
171  //! refered by the variable name used in shaders (string). VBOs are
172  //! pre-allocated with the number of elements (either set from the by
173  //! GLProgram constructor or by the method bind()) but no data are filled,
174  //! this is the job of the developper to do it explicitly.
175  //! - Else, if the VAO was bound previously, nothing is made (list are not
176  //! created/updated).
177  //!
178  //! \note A VAO bound to a different GLProgram cannot be bind to a another
179  //! GLProgram. This is automatically detected and will produce an error. This
180  //! case is refused to avoid to the developer to have a silent error with an
181  //! unexpected behavior.
182  //!
183  //! \param[in] vao the VAO to be bound with this instance of GLProgram.
184  //!
185  //! \return true if the GLVAO has been bound. If bounding cannot be made
186  //! returns false (for one of the following reasons: the GLProgram cannot be
187  //! compiled (syntax errors in shader code) or if the VOA has already been
188  //! bound by another GLProgram (incompatibility).
189  //----------------------------------------------------------------------------
190  inline bool bind(GLVAO& vao)
191  {
192  DEBUG("Binding VAO '%s' to Prog '%s':", vao.cname(), cname());
193 
194  // Try compile the GLProgram if not previously compiled
195  if (unlikely(!isCompiled()))
196  {
197  if (!compile())
198  {
199  ERROR(" Tried to bind VAO '%s' on a non compilable GLProgram '%s'",
200  vao.cname(), cname());
201  return false;
202  }
203  }
204 
205  if (unlikely(0 == vao.prog))
206  {
207  // When binding the VAO to GLProgram for the first time:
208  // populate VBOs and textures in the VAO.
209  DEBUG("%s", " First binding. VBOs will be populated {");
210  initVAO(vao);
211  DEBUG("%s", " } First binding. VBOs will be populated");
212  }
213  else if (unlikely(m_handle != vao.prog))
214  {
215  // Check if VAO has been previously bind by this GLProgram. If not
216  // This is probably an error of the developper.
217  ERROR(" Tried to bind VAO '%s' already bound to another Prog than '%s'",
218  vao.cname(), cname());
219  return false;
220  }
221 
222  // Bind the VAO to the GLProgram. This relationship is now unbreakable
223  m_vao = &vao;
224  return true;
225  }

1051  //! \note Contrary to other GLObject, GLProgram has to perform its
1052  //! setup() before calling activate().

1103  createAllLists();
1104  // Release shaders stored in GPU.
1105  detachAllShaders();
1106  }
