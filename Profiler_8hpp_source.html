41  //! \class Class allowing to measure time of function. It generate a json file
42  //! which can be loaded by the Google browser Chrome. For loading the
43  //! generated file type in the url: "chrome://tracing/" (without '"' chars).
44  //! Beware methods of thi class are not thread safe. Do not use directly this
45  //! class but use macros instead.
46  // ***************************************************************************
47  class Profiler
48  {
49  public:
50 
51  //--------------------------------------------------------------------------
52  //! \brief Profiler can be access from anywhere.
53  //--------------------------------------------------------------------------
54  static Profiler& singleton()
55  {
56  static Profiler instance;
57  return instance;
58  }

60  //--------------------------------------------------------------------------
61  //! \brief Start recording a new session. The previous json file is erased.
62  //! To make things simple, you cannot record several sessions at the same
63  //! time with the same profiler.
64  //--------------------------------------------------------------------------
65  void begin(const char* name, const char* filepath = "profile.json")
66  {
67  m_started = true;
68  m_file.open(filepath);
69  m_file << std::setprecision(3) << std::fixed;
70  header();
71  m_session = name;
72  }
73 
74  //--------------------------------------------------------------------------
75  //! \brief End recording the session.
76  //--------------------------------------------------------------------------
77  void end()
78  {
79  if (!m_started)
80  return ;
81 
82  footer();
83  m_started = false;
84  }
85 
86  //--------------------------------------------------------------------------
87  //! \brief Write time measurement in the json file.
88  //--------------------------------------------------------------------------
89  void write(const char* name, double const start, double const end, size_t const threadID)
90  {
91  if (!m_started)
92  return ;
93 
94  m_file << "{\"cat\":\"function\",\"dur\":" << (end - start);
95  m_file << ",\"name\":\"" << name;
96  m_file << "\",\"ph\":\"X\",\"pid\":0,\"tid\":" << threadID;
97  m_file << ",\"ts\":" << start;
98  m_file << "}," << std::endl;
99  }
100 
101  private:
102 
103  void header()
104  {
105  m_file << "{\"otherData\": {},\"traceEvents\":[" << std::endl;
106  }
107 
108  void footer()
109  {
110  m_file << "{} ]}" << std::endl;
111  }

113  Profiler()

164  Profiler::singleton().write(
165  m_name,
166  static_cast<double>(time_point_cast<nanoseconds>(m_startTimepoint).time_since_epoch().count()) / 1000.0,
167  static_cast<double>(time_point_cast<nanoseconds>(Clock::now()).time_since_epoch().count()) / 1000.0,
168  std::hash<std::thread::id>{}(std::this_thread::get_id())
169  );
170  }
