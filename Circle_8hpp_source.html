37 public:
38 
39  //----------------------------------------------------------------------------
40  //! \brief Constructor. Z-axis aligned tube centered at origin.
41  //! \param radius The radius of the circle.
42  //! \param slices The number of subdivisions around the Z axis.
43  //----------------------------------------------------------------------------
44  Circle(std::string const& name, Material_SP material, float const radius, uint32_t const slices/*, bool const inversed*/)
45  : Shape3D(name, Mode::TRIANGLES, material)
46  {
47  std::vector<float> angle;
48  maths::linspace(0.0f, float(maths::TWO_PI), slices + 1u, angle, true);
49 
50  // Reserve memory
51  vertices().reserve(slices + 2u);
52  normals().reserve(slices + 2u);
53  uv().reserve(slices + 2u);
54  index().reserve(3 * slices);
55 
56  // Constants
57  const float hypothenus = std::sqrt(2.0f * radius * radius);
58  const float hh = radius / hypothenus;
59 
60  // Vertex at the center of the circle
61  vertices().append(Vector3f(0.0f, 0.0f, 0.0f));
62  normals().append(Vector3f(0.0f, 0.0f, 1.0f));
63  uv().append(Vector2f(0.5f, 0.5f));
64 
65  // Vertices constituing the arc
66  for (uint32_t i = 0; i <= slices; ++i)
67  {
68  float c = std::cos(angle[i]);
69  float s = std::sin(angle[i]);
70 
71  vertices().append(Vector3f(radius * c, radius * s, 0.0f));
72  normals().append(Vector3f(hh * c, hh * s, -hh));
73  uv().append(Vector2f((1.0f + c) / 2.0f, (1.0f + s) / 2.0f));
74  }
75 
76  const uint32_t c0 = 0u;
77  const uint32_t i0 = 1u;
78 
79  // Indices for the top cap
80  for (uint32_t i = 0u; i <= slices; ++i)
81  {
82  index().append(c0);
83  index().append(i0 + i);
84  index().append(i0 + i + 1u);
85  }
86  }
87

90  return std::make_shared<Circle>(name, material, radius, slices);
91  }
