31  //! \brief A special node scene allowing to select a single child allowed to
32  //! be traversed: other children are ignored. A SwitchNode3D is not renderable.
33  //! \note this node has been inspired by OpenInventor soSwitch
34  //! https://grey.colorado.edu/coin3d/classSoSwitch.html
35  // ***************************************************************************

37  : public Node3D

51 
52  //--------------------------------------------------------------------------
53  //! \brief Static method allowing to create a switch node.
54  //--------------------------------------------------------------------------
55  static SwitchNode3D_SP create(std::string const& name)
56  {
57  return std::make_shared<SwitchNode3D>(name);

60  //--------------------------------------------------------------------------
61  //! \brief Select the desired child.
62  //!
63  //! The desired child is traversable and other are no longer traversable.
64  //!
65  //! \param nth the number of the desired child.
66  //!
67  //! \note if nth is greater than the number of child. The nth is
68  //! memorized but the effective index is modulo by the number of children.
69  //--------------------------------------------------------------------------
70  void select(size_t const nth)
71  {
72  if (m_designated == nth)
73  return ;
74 
75  m_designated = nth;
76  designated2effective();
77  updateTraversable();
78  }
79 
80  //--------------------------------------------------------------------------
81  //! \brief Return the effective child.
82  //--------------------------------------------------------------------------
83  size_t selected() const
84  {
85  return m_selected;
86  }
87 
88  //--------------------------------------------------------------------------
89  //! \brief Switch to the next child. Start to the first if the last was used.
90  //--------------------------------------------------------------------------
91  void next()
92  {
93  m_designated = (m_designated + 1_z) % m_children.size();
94  m_selected = m_designated;
95  updateTraversable();
96  }
97 
98  //--------------------------------------------------------------------------
99  //! \brief Switch to the previous child. Start to the last if the first was used.
100  //--------------------------------------------------------------------------
101  void previous()
102  {
103  if (unlikely(m_designated == 0_z))
104  m_designated = m_children.size();
105  --m_designated;
106  m_selected = m_designated;
107  updateTraversable();
108  }
109 
110  //--------------------------------------------------------------------------
111  //! \brief Call Node3D::debug() and show additional information.
112  //--------------------------------------------------------------------------
113  virtual void debug() const override
114  {
115  Node3D::debug();
116  std::cout << "Switched to child " << m_selected

119  }

121  protected:
122 
123  //--------------------------------------------------------------------------
124  //! \brief Make the desired child traversable and the others not traversable.
125  //--------------------------------------------------------------------------
126  void updateTraversable()
127  {
128  if (unlikely(m_children.size() == 0_z))
129  return ;
130 
131  for (auto& it: m_children)
132  it->setTraversable(false);
133 
134  m_children[m_selected]->setTraversable(true);
135  }
136 
137  //--------------------------------------------------------------------------
138  //! \brief Callback when a child has been added. Update the traversable
139  //! status for all children.
140  //--------------------------------------------------------------------------
141  virtual void onNodeAdded(Node3D_SP& /*node*/) override
142  {
143  updateTraversable();

145 
146  //--------------------------------------------------------------------------
147  //! \brief Callback when a child has been removed. Update the traversable
148  //! status for all children.
149  //--------------------------------------------------------------------------
150  virtual void onNodeRemoved() override
151  {
152  designated2effective();
153  updateTraversable();
154  }
155 
156  private:
157 
158  //--------------------------------------------------------------------------
159  //! \brief Conver the designated child by the user to the real effective
160  //! selected. Indeed the user may want to select a node not present or when
161  //! removing a child invalidating the selected child. To avoid an out of
162  //! bound error we compute the modulo of the number of children.
163  //--------------------------------------------------------------------------
164  inline void designated2effective()
165  {
166  // m_selected = std::min(m_designated, m_children.size() - 1_z);
167  m_selected = m_designated % m_children.size();
168  }
169 
170  protected:
171 
172  //! \brief Child chosen by the user (index of the std::vector).
173  size_t m_designated = 0u;
174  //! \brief Chosen child converted to a valid std::vector index.
175  size_t m_selected = 0u;
176  };
