65 // *****************************************************************************
66 //! \brief Generic Texture.
67 //!
68 //! A texture is an OpenGL Object that contains one or more images
69 //! that all have the same image format. A texture can be used in two
70 //! ways: either it can be the source of a texture access from a
71 //! Shader, or it can be used as a render target (FrameBuffer). A
72 //! texture can be of dimension 1, 2 or 3.
73 // *****************************************************************************
74 class GLTexture
75  : public GLObject<GLenum>

77 public:

80  //! \brief Constructor. Should be used if you will load texture from

82  //!
83  //! Give a name to the instance. This constructor makes no other
84  //! actions.
85  //!
86  //! \param name the name of this instance used by GLProgram and GLVAO.
87  //! \param target the texture type (GL_TEXTURE_1D .. GL_TEXTURE_3D ...)
88  //----------------------------------------------------------------------------

223  //----------------------------------------------------------------------------
224  //! \brief Apply all settings to the texture.
225  //----------------------------------------------------------------------------
226  void applyTextureParam()
227  {
228  glCheck(glTexParameteri(m_target, GL_TEXTURE_MIN_FILTER,
229  static_cast<GLint>(m_options.minFilter)));
230  glCheck(glTexParameteri(m_target, GL_TEXTURE_MAG_FILTER,
231  static_cast<GLint>(m_options.magFilter)));
232  glCheck(glTexParameteri(m_target, GL_TEXTURE_WRAP_S,
233  static_cast<GLint>(m_options.wrapS)));
234  glCheck(glTexParameteri(m_target, GL_TEXTURE_WRAP_T,
235  static_cast<GLint>(m_options.wrapT)));
236  glCheck(glTexParameteri(m_target, GL_TEXTURE_WRAP_R,
237  static_cast<GLint>(m_options.wrapR)));
