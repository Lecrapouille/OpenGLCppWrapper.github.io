611  virtual bool update() override
612  {

614  switch (res)
615  {
616  case GL_FRAMEBUFFER_COMPLETE:
617  /* success */
618  break;
619  case GL_FRAMEBUFFER_UNDEFINED:
620  throw OpenGLException("Framebuffer '" + name() + "' is undefined");
621  case GL_FRAMEBUFFER_INCOMPLETE_ATTACHMENT:
622  throw OpenGLException("FrameBuffer '" + name() + "' has incomplete attachments");
623  case GL_FRAMEBUFFER_INCOMPLETE_MISSING_ATTACHMENT:
624  throw OpenGLException("Framebuffer '" + name() + "' does not have at least one image attached to it");
625  case GL_FRAMEBUFFER_INCOMPLETE_DRAW_BUFFER:
626  throw OpenGLException("FRAMEBUFFER_INCOMPLETE_DRAW_BUFFER");
627  case GL_FRAMEBUFFER_INCOMPLETE_READ_BUFFER:
628  throw OpenGLException("FRAMEBUFFER_INCOMPLETE_READ_BUFFER");
629  case GL_FRAMEBUFFER_UNSUPPORTED:
630  throw OpenGLException("Framebuffer '" + name() + "' has a combination of internal formats used by attachments is not supported");
631  case 0:
632  default:
