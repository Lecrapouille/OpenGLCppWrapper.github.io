58 // **************************************************************
59 //! \brief Buffer objects are OpenGL objects that store an array of
60 //! unformatted memory allocated by the OpenGL context (aka: the
61 //! GPU). These can be used to store vertex data, pixel data retrieved
62 //! from images or the framebuffer, and a variety of other things.
63 // **************************************************************
64 template<typename T>
65 class GLBuffer
66  : public IGLBuffer,
67  public PendingContainer<T>
68 {
69 public:
70 
71  //! \brief Constructor with the object name
72  explicit GLBuffer(std::string const& name, GLenum const target, BufferUsage const usage)
73  : IGLBuffer(name),
74  PendingContainer<T>(name)
75  {
76  GLObject::m_target = target;
77  m_usage = static_cast<GLenum>(usage);
78  }
79 
80  //! \brief Constructor with the object name and reserved number of
81  //! elements.
82  explicit GLBuffer(std::string const& name, GLenum const target, size_t const init_size, BufferUsage const usage)
83  : IGLBuffer(name),
84  PendingContainer<T>(name, init_size)
85  {
86  GLObject::m_target = target;
87  m_usage = static_cast<GLenum>(usage);
88  }
89 
90  virtual ~GLBuffer() override
91  {
92  destroy();
93  }
94 
95  inline GLenum usage() const
96  {
97  return m_usage;
98  }
99 
100  virtual size_t size() const override
101  {
102  return PendingContainer<T>::size();
103  }
104 
105 private:
106 
107  virtual bool create() override
108  {
109  DEBUG("VBO '%s' create", this->cname());
110  glCheck(glGenBuffers(1, &m_handle));
111  return false;
112  }
113 
114  virtual void activate() override
115  {
116  DEBUG("VBO '%s' activate", this->cname());
117  glCheck(glBindBuffer(m_target, m_handle));
118  }
119 
120  virtual bool setup() override
121  {
122  DEBUG("VBO '%s' setup", this->cname());
123  const GLsizeiptr bytes = static_cast<GLsizeiptr>
124  (PendingContainer<T>::capacity() * sizeof (T));
125  glCheck(glBufferData(m_target, bytes, NULL, m_usage));
126 
127  return false;
128  }
129 
130  virtual inline bool needUpdate() const override
131  {
132  return PendingContainer<T>::hasPendingData();
133  }
134 
135  virtual bool update() override
136  {
137  size_t pos_start, pos_end;
138  PendingContainer<T>::getPendingData(pos_start, pos_end);
139  PendingContainer<T>::clearPending();
140 
141  size_t offset = sizeof (T) * pos_start;
142  size_t nbytes = sizeof (T) * (pos_end - pos_start);
143  glCheck(glBufferSubData(m_target,
144  static_cast<GLintptr>(offset),
145  static_cast<GLsizeiptr>(nbytes),
146  PendingContainer<T>::to_array()));
147 
148  DEBUG("=== FLUSH VBO '%s' to GPU: %zu bytes [%zu .. %zu[",
149  cname(), nbytes, pos_start, pos_end);
150  return false;
151  }
152 
153  virtual void deactivate() override
154  {
155  DEBUG("VBO '%s' deactivate", this->cname());
156  glCheck(glBindBuffer(m_target, 0));
157  }
158 
159  virtual void release() override
160  {
161  DEBUG("VBO '%s' release", this->cname());
162  glCheck(glDeleteBuffers(1, &m_handle));
163  }
164 
165 private:
166 
167  GLenum m_usage;
168 };
