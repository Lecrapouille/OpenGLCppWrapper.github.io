48  //! \brief Empty constructor. It does NOT initialize the matrix for efficiency.

52  //! \brief Constructor with initialization list.
53  Matrix(std::initializer_list<T> initList)
54  {
55  size_t m = std::min(rows * cols, size_t(initList.size()));
56  auto iter = initList.begin();
57  for (size_t i = 0_z; i < m; ++i)
58  {
59  m_data[i] = T(*iter);
60  ++iter;
61  }
62 
63  // Zero-fill any remaining elements.
64  for (size_t i = m; i < rows * cols; ++i)
65  {
66  m_data[i] = maths::zero<T>();
67  }
68  }
69 
70  //! \brief Constructor with an uniform value.
71  explicit Matrix(T a)
72  {
73  size_t i = rows * cols;
74  while (i--)
75  {
76  m_data[i] = a;
77  }
78  }

80  //! \brief Constructor for identity matrix.
81  explicit Matrix(const matrix::MatrixType type)
82  {
83  size_t i;
84 
85  switch (type)
86  {
87  case matrix::Identity:
88  static_assert(rows == cols, "Can't construct identity for a non-square matrix");
89  i = rows * cols; while (i--) { m_data[i] = maths::zero<T>(); }
90  i = cols; while (i--) { m_data[cols * i + i] = maths::one<T>(); }
91  break;
92  default:
93  break;
94  };
95  }

97  //! \brief Constructor by copy.
98  template <typename U, size_t rowsOther, size_t colsOther>
99  explicit Matrix(Matrix<U, rowsOther, colsOther> const &m)
100  {
101  size_t r = std::min(rows, rowsOther);
102  size_t c = std::min(cols, colsOther);
103  for (size_t i = 0_z; i < r; ++i)
104  {
105  for (size_t j = 0_z; j < c; ++j)
106  {
107  (*this)(i, j) = T(m(i, j));
108  }
109  // Zero-fill any remaining cols
110  for (size_t j = c; j < cols; ++j)
111  {
112  (*this)(i, j) = maths::zero<T>();
113  }
114  }
115  // Zero-fill any remaining rows
116  for (size_t i = r * cols; i < rows * cols; ++i)
117  {
118  m_data[i] = maths::zero<T>();
119  }
120  }

333 template <typename T, size_t rows, size_t inner, size_t cols>
334 Matrix<T, rows, cols> operator*(Matrix<T, rows, inner> const &a, Matrix<T, inner, cols> const &b)
335 {
336  Matrix<T, rows, cols> result(maths::zero<T>());
337  for (size_t i = 0_z; i < rows; ++i)
338  for (size_t j = 0_z; j < cols; ++j)

648  template <typename T>
649  Matrix<T, 4_z, 4_z> inverse(Matrix<T, 4_z, 4_z> const& m)
650  {
651  T m00 = m[0][0], m01 = m[0][1], m02 = m[0][2], m03 = m[0][3];
652  T m10 = m[1][0], m11 = m[1][1], m12 = m[1][2], m13 = m[1][3];
653  T m20 = m[2][0], m21 = m[2][1], m22 = m[2][2], m23 = m[2][3];
654  T m30 = m[3][0], m31 = m[3][1], m32 = m[3][2], m33 = m[3][3];
655 
656  T v0 = m20 * m31 - m21 * m30;
657  T v1 = m20 * m32 - m22 * m30;
658  T v2 = m20 * m33 - m23 * m30;
659  T v3 = m21 * m32 - m22 * m31;
660  T v4 = m21 * m33 - m23 * m31;
661  T v5 = m22 * m33 - m23 * m32;
662 
663  T t00 = + (v5 * m11 - v4 * m12 + v3 * m13);
664  T t10 = - (v5 * m10 - v2 * m12 + v1 * m13);
665  T t20 = + (v4 * m10 - v2 * m11 + v0 * m13);
666  T t30 = - (v3 * m10 - v1 * m11 + v0 * m12);
667 
668  T invDet = maths::one<T>() / (t00 * m00 + t10 * m01 + t20 * m02 + t30 * m03);
669  //assert(invDet != maths::zero<T>() && "The matrix cannot be inversed");
670 
671  Matrix<T, 4_z, 4_z> inv;
672  inv[0][0] = t00 * invDet;
673  inv[1][0] = t10 * invDet;
674  inv[2][0] = t20 * invDet;
675  inv[3][0] = t30 * invDet;
676 
677  inv[0][1] = - (v5 * m01 - v4 * m02 + v3 * m03) * invDet;
678  inv[1][1] = + (v5 * m00 - v2 * m02 + v1 * m03) * invDet;
679  inv[2][1] = - (v4 * m00 - v2 * m01 + v0 * m03) * invDet;
680  inv[3][1] = + (v3 * m00 - v1 * m01 + v0 * m02) * invDet;
681 
682  v0 = m10 * m31 - m11 * m30;
683  v1 = m10 * m32 - m12 * m30;
684  v2 = m10 * m33 - m13 * m30;
685  v3 = m11 * m32 - m12 * m31;
686  v4 = m11 * m33 - m13 * m31;
687  v5 = m12 * m33 - m13 * m32;
688 
689  inv[0][2] = + (v5 * m01 - v4 * m02 + v3 * m03) * invDet;
690  inv[1][2] = - (v5 * m00 - v2 * m02 + v1 * m03) * invDet;
691  inv[2][2] = + (v4 * m00 - v2 * m01 + v0 * m03) * invDet;
692  inv[3][2] = - (v3 * m00 - v1 * m01 + v0 * m02) * invDet;
693 
694  v0 = m21 * m10 - m20 * m11;
695  v1 = m22 * m10 - m20 * m12;
696  v2 = m23 * m10 - m20 * m13;
697  v3 = m22 * m11 - m21 * m12;
698  v4 = m23 * m11 - m21 * m13;
699  v5 = m23 * m12 - m22 * m13;
700 
701  inv[0][3] = - (v5 * m01 - v4 * m02 + v3 * m03) * invDet;

703  inv[2][3] = - (v4 * m00 - v2 * m01 + v0 * m03) * invDet;
704  inv[3][3] = + (v3 * m00 - v1 * m01 + v0 * m02) * invDet;

706  return inv;

709  //! \brief This function LU decomposes a given matrix A and stores
710  //! it in two new matricies L and U. It uses the Gaussian
711  //! Elimination with partial pivoting algorithm from Golub & Van
712  //! Loan, Matrix Computations, Algorithm 3.4.1.
713  template <typename T, size_t rows, size_t cols>
714  void LUdecomposition(Matrix<T, rows, cols> const &AA,
715  Matrix<T, rows, cols> &L,
716  Matrix<T, rows, cols> &U,
717  Matrix<T, rows, cols> &P)
718  {
719  size_t i, j;
720 
721  // Set matrices to 0
722  L *= maths::zero<T>();
723  U *= maths::zero<T>();
724 
725  // FIXME: Copy not necessary
726  Matrix<T, rows, cols> A(AA);
727 
728  for (i = 0; i < rows - 1; ++i)
729  {
730  double max = maths::abs(A(i, i));
731  size_t pivot = i;
732 
733  for (j = i + 1_z; j < rows; ++j)
734  {
735  if (maths::abs(A(j, i)) > max)
736  {
737  max = maths::abs(A(j, i));
738  pivot = j;
739  }
740  }
741 
742  if (pivot != i)
743  {
744  matrix::swapRows(A, i, pivot);
745  matrix::swapRows(P, i, pivot);
746  }
747 
748  // ERROR:
749  // -- original code: if (A(i, i) != 0.0)
750  // -- new code which seems to give less good results: if (fabs(A(i, i)) > 0.00001)
751  // we cannot use == with floats or double !!!!
752  if (A(i, i) != maths::zero<T>())
753  {
754  for (j = i + 1_z; j < rows; ++j)
755  {
756  A(j, i) = A(j, i) / A(i, i);
757  for (size_t k = i + 1_z; k < rows; ++k)
758  {
759  A(j, k) = A(j, k) - A(j, i) * A(i, k);
760  }
761  }
762  }
763  }
764  for (i = 0_z; i < rows; ++i)
765  {
766  L(i, i) = maths::one<T>();
767  for (j = 0_z; j < rows; ++j)
768  {
769  if (j < i)
770  {
771  L(i, j) = A(i, j);
772  }
773  else
774  {
775  U(i, j) = A(i, j);
776  }
777  }
778  }
779  }

781  //! \brief This function solves an LU decomposed matrix equation
782  //! LU.x = b.
783  template <typename T, size_t rows, size_t cols>
784  Vector<T, rows> LUsolve(Matrix<T, rows, cols> const &L,
785  Matrix<T, rows, cols> const &U,
786  Matrix<T, rows, cols> const &P,
787  Vector<T, rows> const &b1)
788  {
789  Vector<T, rows> solution(0), y;
790 
791  // Apply permutation
792  Vector<T, rows> b(P * b1);
793 
794  // y = U.x, thus Ly = b
795  // solve for y by forward substitution
796  y[0] = b[0] / L(0, 0);
797  for (size_t i = 1_z; i < rows; ++i)
798  {
799  y[i] = b[i] / L(i, i);
800  for (size_t j = 0_z; j < i; ++j)
801  {
802  y[i] -= (L(i, j) * y[j] / L(i, i));
803  }
804  }
805 
806  // U.x = y
807  // Solve for x by backward substitution
808  size_t r = rows - 1_z;
809  solution[r] = y[r] / U(r, r);
810 
811  size_t i = r;
812  while (i--)
813  {
814  solution[i] = y[i] / U(i, i);
815  for (size_t j = i + 1u; j < rows; j++)
816  {
817  solution[i] -= (U(i, j) * solution[j] / U(i, i));
818  }
819  }
820  return solution;
821  }
