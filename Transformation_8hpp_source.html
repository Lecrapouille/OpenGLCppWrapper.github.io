57  //! \brief angle in radian
58  template<typename T>
59  Matrix<T, 4_z, 4_z> rotate(Matrix<T, 4_z, 4_z> const& m, T const angle, Vector<T, 3_z> const& v)
60  {
61  T const c = std::cos(angle);
62  T const s = std::sin(angle);
63 
64  Vector<T, 3_z> axis(vector::normalize(v));
65  Vector<T, 3_z> temp((maths::one<T>() - c) * axis);
66  Matrix<T, 4_z, 4_z> rotate;
67 
68  rotate(0, 0) = c + temp[0] * axis[0];
69  rotate(0, 1) = temp[0] * axis[1] + s * axis[2];
70  rotate(0, 2) = temp[0] * axis[2] - s * axis[1];
71 
72  rotate(1, 0) = temp[1] * axis[0] - s * axis[2];
73  rotate(1, 1) = c + temp[1] * axis[1];
74  rotate(1, 2) = temp[1] * axis[2] + s * axis[0];
75 
76  rotate(2, 0) = temp[2] * axis[0] + s * axis[1];
77  rotate(2, 1) = temp[2] * axis[1] - s * axis[0];
78  rotate(2, 2) = c + temp[2] * axis[2];
79 
80  Matrix<T, 4_z, 4_z> M;
81  M[0] = m[0] * rotate(0, 0) + m[1] * rotate(0, 1) + m[2] * rotate(0, 2);
82  M[1] = m[0] * rotate(1, 0) + m[1] * rotate(1, 1) + m[2] * rotate(1, 2);
83  M[2] = m[0] * rotate(2, 0) + m[1] * rotate(2, 1) + m[2] * rotate(2, 2);
84  M[3] = m[3];
85 
86  return M;
87  }

91  {

103  // *************************************************************************************************
104  //! \brief Replace gluPerspective(). Set the frustum to perspective mode.
105  //! \param fovY - Field of vision in radians in the y direction.
106  //! \param aspect - Aspect ratio of the viewport.
107  //! \param zNear - The near clipping distance.
108  //! \param zFar - The far clipping distance.
109  // *************************************************************************************************
110  template<typename T>
111  Matrix<T, 4_z, 4_z> perspective(T const fovY, T const aspect, T const zNear, T const zFar)
112  {
113  assert(maths::abs(aspect - std::numeric_limits<T>::epsilon()) > static_cast<T>(0));
114 
115  T const tanHalfFovY = std::tan(fovY / T(2));
116  Matrix<T, 4_z, 4_z> M(0);
117 
118  M(0, 0) = maths::one<T>() / (aspect * tanHalfFovY);
119  M(1, 1) = maths::one<T>() / (tanHalfFovY);
120  M(2, 3) = -maths::one<T>();
121  M(2, 2) = -(zFar + zNear) / (zFar - zNear);
122  M(3, 2) = -(T(2) * zFar * zNear) / (zFar - zNear);
123 
124  return M;
125  }

127  // *************************************************************************************************
128  //! \brief Camera looks at.
129  //! \param position camera position.
130  //! \param target the target position the camera is looking at.
131  //! \param up upwards direction of the camera.
132  //!
133  //! | Rx Ry Rz 0 | | 1 0 0 -Px |
134  //! LookAt = | Ux Uy Uz 0 | x | 0 1 0 -Py |
135  //! | Dx Dy Dz 0 | | 0 0 1 -Pz |
136  //! | 0 0 0 1 | | 0 0 0 1 |
137  //!
138  //! Where U is the up vector, D is the direction vector, P is the camera's
139  //! position vector and R is the right vector (= cross product between U and D).
140  // *************************************************************************************************
141  template<typename T>
142  Matrix<T, 4_z, 4_z> lookAt(Vector<T, 3_z> const& position,
143  Vector<T, 3_z> const& target,
144  Vector<T, 3_z> const& upwards)
145  {
146  Vector<T, 3_z> const direction(vector::normalize(target - position));
147  Vector<T, 3_z> const right(vector::normalize(vector::cross(direction, upwards)));
148  Vector<T, 3_z> const up(vector::cross(right, direction));
149  Matrix<T, 4_z, 4_z> M(matrix::Identity);
150 
151  M(0, 0) = right.x;
152  M(1, 0) = right.y;
153  M(2, 0) = right.z;
154  M(0, 1) = up.x;
155  M(1, 1) = up.y;
156  M(2, 1) = up.z;
157  M(0, 2) = -direction.x;
158  M(1, 2) = -direction.y;
159  M(2, 2) = -direction.z;
160  M(3, 0) = -(vector::dot(right, position));
161  M(3, 1) = -(vector::dot(up, position));
162  M(3, 2) = vector::dot(direction, position);
163 
164  return M;
165  }
