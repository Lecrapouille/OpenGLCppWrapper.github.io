117  //----------------------------------------------------------------------------
118  //! \brief Read the whole shader code from the given ascii file.
119  //! Note: the shader code compilation is delayed. The code is only
120  //! store in the member variable.
121  //! \param[in] path the path of the shader code.
122  //! \return true if the whole file has been read, else return false.
123  //! An error message is set and can be read through getError().
124  //----------------------------------------------------------------------------
125  bool fromFile(std::string const& path)
126  {
127  throw_if_already_compiled();
128 
129  std::string msg;
130  std::ifstream infile;
131 
132  infile.open(path, std::ifstream::in);
133  if (infile.fail())
134  {
135  msg = "Failed opening file '" + path + "'. Reason was '" + strerror(errno) + "'";
136  goto l_error;
137  }
138 
139  if (!GLShader::resize(infile, m_shader_code))
140  {
141  msg = "Failed reading file '" + path + "'. Reason was 'cannot get the file size'";
142  goto l_error;
143  }
144 
145  if (!infile.read(&m_shader_code[0U], static_cast<std::streamsize>(m_shader_code.size())))
146  {
147  msg = "Failed reading file '" + path + "'. Reason was '" + strerror(errno) + "'";
148  goto l_error;
149  }
150 
151  DEBUG("From file '%s' %s: '%s'", cname(), type(), m_shader_code.c_str());
152  return true;
153 
154 l_error:
155  concatError(msg);
156  ERROR("%s", msg.c_str());
157  return false;
158  }

265  //----------------------------------------------------------------------------
266  //! \brief Compile the shader code in the GPU.
267  //! \return true if the compilation succeeded. Else return false.
268  //! An error message is set and can be read through getError().
269  //----------------------------------------------------------------------------
270  virtual bool setup() override
271  {
272  DEBUG(" Shader '%s' setup", cname());
273  if (likely((loaded() && !isCompiled())))
274  {
275  char const *source = m_shader_code.c_str();
276  GLint length = static_cast<GLint>(m_shader_code.size());
277  glCheck(glShaderSource(m_handle, 1, &source, &length));
278  glCheck(glCompileShader(m_handle));
279  m_compiled = checkCompilationStatus(m_handle);
280  }
281  else
282  {
283  std::string msg =
284  " Could not compile the shader named '" + name() + "'. Reason was ";
285  msg += (loaded()) ? "'already compiled'" : "'compilation errored'";
286  concatError(msg);
287  ERROR("%s", msg.c_str());
288  }
289  DEBUG(" Shader '%s' compiled? %d", cname(), m_compiled);
290  return !m_compiled;
291  }

313  //----------------------------------------------------------------------------
314  //! \brief Check if a shader script has been correctly compiled.
315  //!
316  //! An error message is set in m_error_msg and which can be read through
317  //! getError().
318  //!
319  //! \param[in] obj the identifer of the loaded shader.
320  //! \return true if the shader has been compiled with success else return false.
321  //----------------------------------------------------------------------------
322  bool checkCompilationStatus(GLuint obj)
323  {
324  GLint status;
325 
326  glCheck(glGetShaderiv(obj, GL_COMPILE_STATUS, &status));
327  if (GL_FALSE == status)
328  {
329  GLint length;
330  glCheck(glGetShaderiv(obj, GL_INFO_LOG_LENGTH, &length));
331  std::vector<char> log(static_cast<size_t>(length));
332  glCheck(glGetShaderInfoLog(obj, length - 1, &length, &log[0U]));
333  concatError(&log[0U]);
334  ERROR("Failed compiling '%s'. Reason was '%s'", cname(), m_error_msg.c_str());
335  }
336  else
337  {
338  m_error_msg.clear();
339  }
340  return status;
341  }
