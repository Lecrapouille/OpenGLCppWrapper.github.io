127  //----------------------------------------------------------------------------
128  //! \brief Load a jpeg, png, bmp ... file as texture.
129  //!

131  //! \param rename if set to true then the name of this instance (set

137  inline bool load(const char *const filename)

198  //----------------------------------------------------------------------------
199  //! \brief Specify to OpenGL a two-dimensional texture image.
200  //----------------------------------------------------------------------------
201  inline void specifyTexture2D() const
202  {
203  // Note: is allowed this case:
204  // m_width != 0 and m_height != 0 and buffer == nullptr
205  // This will reserve the buffer size.
206  glCheck(glTexImage2D(m_target, 0,
207  static_cast<GLint>(m_gpuPixelFormat),
208  static_cast<GLsizei>(m_width),
209  static_cast<GLsizei>(m_height),
210  0,
211  static_cast<GLenum>(m_cpuPixelFormat),
212  static_cast<GLenum>(m_options.pixelType),
213  m_buffer.to_array()));
214  }

216  //----------------------------------------------------------------------------
217  //! \brief Apply OpenGL texture settings.
218  //----------------------------------------------------------------------------
219  virtual bool setup() override
220  {
221  DEBUG("Texture '%s' setup", cname());
222 
223  // Note: m_buffer can nullptr
224  if (unlikely(!loaded()))
225  {
226  ERROR("Cannot setup texture '%s'. Reason 'Data not yet loaded'", cname());
227  return true;
228  }
229 
230  applyTextureParam();
231  specifyTexture2D();
232  return false;
233  }
234 
235  //----------------------------------------------------------------------------
236  //! \brief Upload dirty CPU data to the GPU.
237  //----------------------------------------------------------------------------
238  virtual bool update() override
239  {
240  //TODO
241 # pragma GCC diagnostic push
242 # pragma GCC diagnostic ignored "-Wsign-conversion"
243 # pragma GCC diagnostic ignored "-Wconversion"
244  DEBUG("Texture '%s' update", cname());
245 
246  size_t start, stop;
247  m_buffer.getPendingData(start, stop);
248 
249  start = start / 4; //TODO 4 because RGBA
250  stop = stop / 4;
251  const GLint x = start / m_width;
252  const GLint y = start % m_width;
253  const GLsizei width = (stop / m_width) - x;
254  const GLsizei height = (stop % m_width) - y;
255 
256  DEBUG("=== FLUSH Texture2D '%s' to GPU: (%zu,%zu) --> ((%d,%d), (%d,%d))",
257  cname(), start, stop, x, y, width, height);
258 
259  // FIXME: not working if width and height are not the txture size
260  glCheck(glBindTexture(m_target, m_handle));
261  glCheck(glTexSubImage2D(m_target, 0, x, y, width, height,
262  static_cast<GLenum>(m_cpuPixelFormat),
263  static_cast<GLenum>(m_options.pixelType),
264  m_buffer.to_array()));
265 
266  m_buffer.clearPending();
267  return false;
268 # pragma GCC diagnostic pop
269  };
