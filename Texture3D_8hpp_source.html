44 class GLTextureCube: public GLTexture

46 public:
47 
48  //----------------------------------------------------------------------------
49  //! \brief Constructor.
50  //!
51  //! Give a name to the instance. This constructor makes no other
52  //! actions.
53  //!
54  //! \param name the name of this instance used by GLProgram and GLVAO.
55  //----------------------------------------------------------------------------
56  GLTextureCube(std::string const& name)
57  : GLTexture(3u, name, GL_TEXTURE_CUBE_MAP)

60  virtual ~GLTextureCube()

63  //----------------------------------------------------------------------------
64  //! \brief \brief Allow to know if data have been transfered into
65  //! the CPU memory.
66  //!
67  //! \return true if the 6 textures 2D (one by face of the cube) have
68  //! been loaded.
69  //----------------------------------------------------------------------------
70  virtual inline bool loaded() const override
71  {
72  for (uint8_t i = 0; i < 6u; ++i)
73  {
74  if (unlikely(false == m_textures[i]->loaded()))
75  return false;
76  }
77  return true;
78  }
79 
80  //----------------------------------------------------------------------------
81  //! \brief Load a texture 2D at the given location on the cube.
82  //!
83  //! \note to be a valid GLTextureCube all faces of the cube shall
84  //! have loaded (so this method has to be called 6 times with the
85  //! correct \p target).
86  //!
87  //! \param target select one of the 6 faces of the cube.
88  //! \param filename the path of a jpeg or bmp or png file.
89  //!
90  //! \return false if the texture failed to be loaded.
91  //----------------------------------------------------------------------------
92  bool load(CubeMap const target, const char *const filename)
93  {
94  const size_t index = static_cast<size_t>(target) -
95  static_cast<size_t>(GL_TEXTURE_CUBE_MAP_POSITIVE_X);
96 
97  DEBUG("TextureCube '%s' %zu load bitmap '%s'", cname(), index, filename);
98  return m_textures[index]->load(filename);
99  }
100 
101 private:
102 
103  //----------------------------------------------------------------------------
104  //! \brief Apply OpenGL texture settings.
105  //----------------------------------------------------------------------------
106  virtual bool setup() override
107  {
108  // Note: m_buffer can nullptr
109  m_depth = computeDepth();
110  if (6u != m_depth)
111  {
112  ERROR("Cannot setup texture '%s'. Reason 'Data not yet loaded'", cname());
113  return true;
114  }
115 
116  for (uint8_t i = 0; i < 6u; ++i)
117  {
118  m_textures[i]->m_target = targets(i);
119  m_textures[i]->options(m_options);
120  m_textures[i]->specifyTexture2D();
121  }
122  applyTextureParam();
123  return false;
124  }
125 
126  //----------------------------------------------------------------------------
127  //! \brief Upload dirty CPU data to the GPU.
128  //----------------------------------------------------------------------------
129  virtual bool update() override
130  {
131  glCheck(glBindTexture(m_target, m_handle));
132  for (uint8_t i = 0; i < 6u; ++i)
133  {
134  m_textures[i]->update();
135  }
136  return false;
137  }
138 
139  //----------------------------------------------------------------------------
140  //! \brief
141  //----------------------------------------------------------------------------
142  uint8_t computeDepth()
143  {
144  uint8_t depth = 0u;
145  for (uint8_t i = 0; i < 6u; ++i)
146  {
147  if (likely(m_textures[i]->loaded()))
148  ++depth;
149  }
150  return depth;
151  }
152 
153  //----------------------------------------------------------------------------
154  //! \brief
155  //----------------------------------------------------------------------------
156  static GLenum targets(const uint8_t index)
157  {
158  static constexpr std::array<GLenum, 6> targets =
159  {{
160  GL_TEXTURE_CUBE_MAP_POSITIVE_X,
161  GL_TEXTURE_CUBE_MAP_NEGATIVE_X,
162  GL_TEXTURE_CUBE_MAP_POSITIVE_Y,
163  GL_TEXTURE_CUBE_MAP_NEGATIVE_Y,
164  GL_TEXTURE_CUBE_MAP_POSITIVE_Z,
165  GL_TEXTURE_CUBE_MAP_NEGATIVE_Z
166  }};
167  return targets[index];
168  }
169 
170 private:
171 
172  //----------------------------------------------------------------------------
173  //! \brief Create the 6 textures use for mapping the cube.
174  //!
175  //! \note: unique_ptr is to avoid calling the deleted GLTexture2D copy
176  //! constructor (GLObject are non copyable).
177  //----------------------------------------------------------------------------
178  std::array<GLTexture2D_UP, 6> m_textures
179  {{
180  std::make_unique<GLTexture2D>("CUBEMAP_POSITIVE_X"),
181  std::make_unique<GLTexture2D>("CUBEMAP_NEGATIVE_X"),
182  std::make_unique<GLTexture2D>("CUBEMAP_POSITIVE_Y"),
183  std::make_unique<GLTexture2D>("CUBEMAP_NEGATIVE_Y"),
184  std::make_unique<GLTexture2D>("CUBEMAP_POSITIVE_Z"),
185  std::make_unique<GLTexture2D>("CUBEMAP_NEGATIVE_Z"),
186  }};
187 };

190 //! \brief A 3D Texture.
191 // *****************************************************************************
192 class GLTexture3D: public GLTexture
193 {
194 public:
195 
196  //----------------------------------------------------------------------------
197  //! \brief
198  //----------------------------------------------------------------------------
199  GLTexture3D(std::string const& name)
200  : GLTexture(3u, name, GL_TEXTURE_3D)
201  {}
202 
203  //----------------------------------------------------------------------------
204  //! \brief
205  //----------------------------------------------------------------------------
206  bool load(std::vector<std::string> const& filenames)
207  {
208  uint32_t width = 0;
209  uint32_t height = 0;
210  uint32_t prevWidth = 0;
211  uint32_t prevHeight = 0;
212  size_t depth = filenames.size();
213  SOIL soil(m_cpuPixelFormat);
214 
215  m_buffer.clear();
216  for (size_t i = 0; i < depth; ++i)
217  {
218  DEBUG("Texture3D '%s' load bitmap '%s'", cname(), filenames[i].c_str());
219 
220  // Load a Texture2D and pack it subsequently into a large 2D texture
221  width = height = 0;
222  if (unlikely(!soil.load(filenames[i].c_str(), m_buffer, width, height)))
223  return false;
224 
225  // Check consistency of Texture2D dimension
226  if ((i != 0) && ((prevWidth != width) || (prevHeight != height)))
227  {
228  ERROR("Failed picture file '%s' has not correct dimension %ux%u",
229  filenames[i].c_str(), prevWidth, prevHeight);
230  return false;
231  }
232 
233  prevWidth = width;
234  prevHeight = height;
235  }
236 
237  // Success
238  m_width = static_cast<uint32_t>(width);
239  m_height = static_cast<uint32_t>(height);
240  m_depth = static_cast<uint8_t>(depth);
241  return true;
242  }
243 
244 private:
245 
246  //----------------------------------------------------------------------------
247  //! \brief Specify to OpenGL a three-dimensional texture image.
248  //----------------------------------------------------------------------------
249  inline void specifyTexture3D() const
250  {
251  glCheck(glTexImage3D(m_target, 0,
252  static_cast<GLint>(m_gpuPixelFormat),
253  static_cast<GLsizei>(m_width),
254  static_cast<GLsizei>(m_height),
255  static_cast<GLsizei>(m_depth),
256  0,
257  static_cast<GLenum>(m_cpuPixelFormat),
258  static_cast<GLenum>(m_options.pixelType),
259  m_buffer.to_array()));
260  }
261 
262  //----------------------------------------------------------------------------
263  //! \brief Apply OpenGL texture settings.
264  //----------------------------------------------------------------------------
265  virtual bool setup() override
266  {
267  DEBUG("Texture '%s' setup", cname());
268 
269  // Note: m_buffer can nullptr
270  if (unlikely(!loaded()))
271  {
272  ERROR("Cannot setup texture '%s'. Reason 'Data not yet loaded'", cname());
273  return true;
274  }
275 
276  // Data is aligned in byte order
277  glCheck(glPixelStorei(GL_UNPACK_ALIGNMENT, 1));
278 
279  applyTextureParam();
280  specifyTexture3D();
281 
282  return false;
283  }
284

287  //! \todo To be implemeted FIXME
288  //----------------------------------------------------------------------------
289  virtual bool update() override
290  {
291  DEBUG("Texture '%s' update", cname());
292  return false;
293  };
294 };
