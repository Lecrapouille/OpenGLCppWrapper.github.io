79  //! \brief Load picture file (jpg, png ...)
80  //! \param[in] filename the path of the picture file.
81  //! \param[inout] data the pending container holding the texture (CPU side)
82  //! \param[out] width return the texture width (in pixels)
83  //! \param[out] height return the texture height (in pixel)
84  //! \return true if the loading ends with success, else return false.
85  //----------------------------------------------------------------------------
86  bool load(const char *const filename, TextureData& data,
87  uint32_t& width, uint32_t& height)
88  {
89  DEBUG("Loading texture '%s'", filename);
90 
91  if (unlikely(nullptr == filename))
92  return false;
93  if (unlikely(!m_isvalid))
94  return false;
95 
96  // Load the image as a C array.
97  int w, h;
98  unsigned char* image = SOIL_load_image(filename, &w, &h, 0,
99  static_cast<int>(m_soilformat));
100  if (likely(nullptr != image))
101  {
102  // Use the max because with framebuffer we can resize texture
103  width = std::max(width, static_cast<uint32_t>(w)); // FIXME
104  height = std::max(height, static_cast<uint32_t>(h));
105 
106  // Convert it as std::vector
107  size_t size = static_cast<size_t>(w * h) * m_nbpixel * sizeof(unsigned char);
108  data.append(image, size); // FIXME: not working with preallocated size
109  SOIL_free_image_data(image);
110  DEBUG("Successfully loaded %ux%u texture '%s'", width, height, filename);
111  return true;
112  }
113  else
114  {
115  ERROR("Failed loading picture file '%s'. Reason was '%s'",
116  filename, SOIL_last_result());
117  width = height = 0;
118  data.clear();
119  return false;
120  }
121  }

123  bool save(const char *const filename, TextureData const& data,
124  uint32_t const width, uint32_t const height)
125  {
126  const unsigned char* buffer = data.to_array();
127  if (unlikely(nullptr == buffer))
128  {
129  ERROR("Cannot save a texture with no data");
130  return false;
131  }
132 
133  std::string ext = getExtension(filename);
134  if (ext == "bmp")
135  m_soilformat = SOIL_SAVE_TYPE_BMP;
136  else if (ext == "tga")
137  m_soilformat = SOIL_SAVE_TYPE_TGA;
138  else if (ext == "dds")
139  m_soilformat = SOIL_SAVE_TYPE_DDS;
140  else
141  {
142  ERROR("Cannot save a texture into the given file format '%s'",
143  ext.c_str());
144  return false;
145  }
146 
147  bool res = !!SOIL_save_image(filename, m_soilformat,
148  static_cast<int>(width),
149  static_cast<int>(height),
150  static_cast<int>(m_nbpixel), buffer);
151  if (unlikely(!res))
152  ERROR("Failed saving the texture in the file '%s'. Reason was '%s'",
153  filename, SOIL_last_result());
154 
155  return res;
156  }

160  std::string getExtension(std::string const& path)
161  {
162  std::string::size_type pos = path.find_last_of(".");
163  if (pos != std::string::npos)
164  {
165  std::string ext = path.substr(pos + 1, std::string::npos);
166 
167  // Ignore the ~ in the extension (ie. foo.txt~)
168  if ('~' == ext.back())
169  ext.pop_back();
170 
171  // Get the file extension in lower case
172  std::transform(ext.begin(), ext.end(), ext.begin(), ::tolower);
173  return ext;
174  }
175  return "";
176  }
