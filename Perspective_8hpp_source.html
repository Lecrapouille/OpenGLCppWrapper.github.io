186  void incrAngles(float const yaw, float const pitch) // radians
187  {
188  m_pitch += pitch;

199  void initAngles()
200  {
201  Vector3f &f = m_front;
202  float length = vector::norm(m_front);
203  m_yaw = 360 + maths::toDegree(atan2f(f.z, f.x));
204  m_pitch = maths::toDegree(asinf(f.y / length));
205 
206  updateViewMatrix();
207  }
208 
209  void updateStates()
210  {
211  m_direction.x = cosf(maths::toRadian(m_yaw)) * cosf(maths::toRadian(m_pitch));
212  m_direction.y = sinf(maths::toRadian(m_pitch));
213  m_direction.z = sinf(maths::toRadian(m_yaw)) * cosf(maths::toRadian(m_pitch));
214  m_front = vector::normalize(m_direction);
215  m_right = vector::normalize(vector::cross(m_front, m_worldUp));
216  m_up = vector::normalize(vector::cross(m_right, m_front));
217 
218  updateViewMatrix();
219  }
