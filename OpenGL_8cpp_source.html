76 {
77  GLenum id;
78  const char* error;
79 
80  while ((id = glGetError()) != GL_NO_ERROR)
81  {
82  switch (id)
83  {
84  case GL_INVALID_OPERATION:
85  error = "GL_INVALID_OPERATION";
86  break;
87  case GL_INVALID_ENUM:
88  error = "GL_INVALID_ENUM";
89  break;
90  case GL_INVALID_VALUE:
91  error = "GL_INVALID_VALUE";
92  break;
93  case GL_OUT_OF_MEMORY:
94  error = "GL_OUT_OF_MEMORY";
95  break;
96  case GL_INVALID_FRAMEBUFFER_OPERATION:
97  error = "GL_INVALID_FRAMEBUFFER_OPERATION";
98  break;
99  default:
100  error = "UNKNOWN";
101  break;
102  }
103 
104  // Do not use directly LOG macros because it will catch this
105  // filename and its line instead of the faulty file/line which
106  // produced the OpenGL error.
107  errout(stderr, "GLERR:", filename, line, "Failed executing '%s'. Reason is '%s'",
108  expression, error);
