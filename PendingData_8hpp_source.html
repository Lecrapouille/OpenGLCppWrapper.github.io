36 //! \class PendingData PendingData.hpp
37 //!
38 //! \brief Define class tracks the smallest contiguous area in a container
39 //! having been modified (aka updated aka dirty). This class cannot be use
40 //! alone. Use it with inheritance.
41 //!
42 //! This class is used in this API for knowing dirty CPU data that needs to be
43 //! updated to the GPU.
44 // ***************************************************************************
45 class PendingData
46 {
47 public:
48 
49  //--------------------------------------------------------------------------
50  //! \brief When the container is empty place the smallest contiguous area of
51  //! dirty elements is not defined. So use -1 instead of 0. It is equivalent
52  //! of std::string::npos.
53  //--------------------------------------------------------------------------
54  static constexpr size_t npos = static_cast<size_t>(-1);
55 
56 protected:
57 
58  //--------------------------------------------------------------------------
59  //! \brief Make this class abstract to forbid to use it as standalone.
60  //! Derivate this class to use it.
61  //--------------------------------------------------------------------------
62  virtual ~PendingData() {};
63 
64 public:
65 
66  //--------------------------------------------------------------------------
67  //! \brief Empty constructor with no pending data.
68  //--------------------------------------------------------------------------
69  PendingData()
70  {}
71 
72  //--------------------------------------------------------------------------
73  //! \brief Constructor with an already known number of dirty elements
74  //! (starting from the first).
75  //--------------------------------------------------------------------------
76  PendingData(size_t const nb_elt)
77  {
78  clearPending(nb_elt);
79  }
80 
81  //--------------------------------------------------------------------------
82  //! \brief Return a boolean indicating if some elements of the container
83  //! are dirty (modified).
84  //!
85  //! \return true if data are dirty.
86  //--------------------------------------------------------------------------
87  inline bool hasPendingData() const
88  {
89  return (npos != m_pending_start);
90  }
91 
92  //--------------------------------------------------------------------------
93  //! \brief Return the smallest contiguous area of the container having
94  //! dirty elements.
95  //!
96  //! In the case there is no pending data, the return area bound is logicaly
97  //! undefined but we return pos_start and pos_end are set to \c npos. You
98  //! can call hasPendingData() before calling this method.
99  //--------------------------------------------------------------------------
100  void getPendingData(size_t& pos_start, size_t& pos_end) const
101  {
102  pos_start = m_pending_start;
103  pos_end = m_pending_end;
104  }
105 
106  //--------------------------------------------------------------------------
107  //! \brief Return the smallest contiguous area of the container having
108  //! dirty elements.
109  //!
110  //! In the case there is no pending data, the return area bound is logicaly
111  //! undefined but we return pos_start and pos_end are set to \c npos. You
112  //! can call hasPendingData() before calling this method.
113  //!
114  //! \return {pos_start, pos_end}
115  //--------------------------------------------------------------------------
116  std::pair<size_t, size_t> getPendingData() const
117  {
118  return { m_pending_start, m_pending_end };
119  }
120 
121  //--------------------------------------------------------------------------
122  //! \brief Reset the smallest contiguous area of the container having
123  //! dirty elements.
124  //!
125  //! Call this the smallest contiguous area have been updated and therefore
126  //! when there is no more dirty data.
127  //--------------------------------------------------------------------------
128  void clearPending()
129  {
130  m_pending_start = npos;
131  m_pending_end = npos;
132  }
133 
134  //--------------------------------------------------------------------------
135  //! \brief Reset the smallest contiguous area of the container having dirty
136  //! elements and tag the \c nb_elt first elements as dirty. Use this method
137  //! from constructors.
138  //--------------------------------------------------------------------------
139  void clearPending(size_t const nb_elt)
140  {
141  if (0_z == nb_elt)
142  {
143  m_pending_start = npos;
144  m_pending_end = npos;
145  }
146  else
147  {
148  m_pending_start = 0_z;
149  m_pending_end = nb_elt;
150  }
151  }
152 
153  //--------------------------------------------------------------------------
154  //! \brief Update the area of dirty elements with a new range.
155  //--------------------------------------------------------------------------
156  void tagAsPending(size_t const pos_start, size_t const pos_end)
157  {
158  if (likely(hasPendingData()))
159  {
160  m_pending_start = std::min(m_pending_start, pos_start);
161  m_pending_end = std::max(m_pending_end, pos_end);
162  }
163  else
164  {
165  m_pending_start = pos_start;
166  m_pending_end = pos_end;
167  }
168  }
169 
170  //--------------------------------------------------------------------------
171  //! \brief Update the area of dirty elements by passing the newly dirty
172  //! element.
173  //--------------------------------------------------------------------------
174  inline void tagAsPending(size_t const pos)
175  {
176  if (likely(hasPendingData()))
177  {
178  m_pending_start = std::min(m_pending_start, pos);
179  m_pending_end = std::max(m_pending_end, pos);
180  }
181  else
182  {
183  m_pending_start = pos;
184  m_pending_end = pos;
185  }
186  }
187 
188 protected:
189 
190  //! Tracks the staring bound of elementsthat have been changed.
191  size_t m_pending_start = npos;
192 
193  //! Tracks the ending bound of elementsthat have been changed.
194  size_t m_pending_end = npos;
195 };
