32 //! \brief Create a tube shape. Tube is a generic shape that can create Cylinder
33 //! Cone and Pyramid.
34 //! \ingroup Geometry
35 // *****************************************************************************
36 class Tube: public Shape3D
37 {
38 public:
39 
40  //----------------------------------------------------------------------------
41  //! \brief Constructor. Z-axis aligned tube centered at origin.
42  //! \param[in] top_radius The radius at the top of the tube.
43  //! \param[in] base_radius The radius at the base of the tube.
44  //! \param[in] height The height of the tube.
45  //! \param[in] slices The number of subdivisions around the Z axis.
46  //----------------------------------------------------------------------------
47  Tube(std::string const& name, Material_SP material, float const top_radius, float const base_radius,
48  float const height, uint32_t const slices)
49  : Shape3D(name, Mode::TRIANGLES, material)
50  {
51  // Create caps ?
52  const bool base_caps = base_radius > 0.0f;
53  const bool top_caps = top_radius > 0.0f;
54 
55  std::vector<float> angle;
56  maths::linspace(0.0f, float(maths::TWO_PI), slices + 1u, angle, true);
57 
58  std::vector<float> texture;
59  maths::linspace(0.0f, 1.0f, slices + 1u, texture, true);
60 
61  // Reserve memory
62  // (slices + 1u) vertices for by circle
63  // +1 by radius if cap it
64  // x2 because of top and base radius
65  uint32_t nb_points = 2u * (slices + 1u);
66  vertices().resize(nb_points);
67  normals().resize(nb_points);
68  uv().resize(nb_points);
69  nb_points = 6u * (slices + 1u)
70  + (base_caps ? 3u * slices : 0u)
71  + (top_caps ? 3u * slices : 0u);
72  index().reserve(nb_points);
73 
74  // Constants
75  const float abs_top_radius = maths::abs(top_radius);
76  const float abs_base_radius = maths::abs(base_radius);
77  const float h2 = height / 2.0f;
78  const float r = abs_top_radius - abs_base_radius;
79  const float hypothenus = std::sqrt(r * r + height * height);
80  const float hh = height / hypothenus;
81  const float rh = -r / hypothenus;
82 
83  // First index for top and base radius
84  uint32_t const i0 = 0u;
85  uint32_t const i1 = i0 + slices + 1u;
86 
87  for (uint32_t i = 0; i <= slices; ++i)
88  {
89  float c = std::cos(angle[i]);
90  float s = std::sin(angle[i]);
91 
92  // Top of the tube
93  vertices(i0 + i) = Vector3f(abs_top_radius * c, abs_top_radius * s, h2);
94  normals(i0 + i) = Vector3f(hh * c, hh * s, rh);
95  uv(i0 + i) = Vector2f(texture[i], 0.0f);
96 
97  // Bottom of the tube
98  vertices(i1 + i) = Vector3f(abs_base_radius * c, abs_base_radius * s, -h2);
99  normals(i1 + i) = Vector3f(hh * c, hh * s, rh);
100  uv(i1 + i) = Vector2f(texture[i], 1.0f);
101 
102  // Indices for the tube
103  index().append(i0 + i); index().append(i0 + i + 1u); index().append(i1 + i);
104  index().append(i1 + i); index().append(i1 + i + 1u); index().append(i0 + i + 1u);
105  }
106 
107  if (top_caps)
108  {
109  // c0: Vertex at the center of the top cap
110  uint32_t const c0 = uint32_t(vertices().size());
111  vertices().append(Vector3f(0.0f, 0.0f, h2));
112  normals().append(Vector3f(0.0f, 0.0f, 1.0f));
113  uv().append(Vector2f(0.5f, 0.5f));
114 
115  for (uint32_t i = 0; i < slices; ++i)
116  {
117  index().append(c0);
118  index().append(i0 + i);
119  index().append(i0 + i + 1u);
120  }
121  }
122 
123  if (base_caps)
124  {
125  // c1: Vertex at the center of the bottom cap
126  uint32_t const c1 = uint32_t(vertices().size());
127  vertices().append(Vector3f(0.0f, 0.0f, -h2));
128  normals().append(Vector3f(0.0f, 0.0f, -1.0f));
129  uv().append(Vector2f(0.5f, 0.5f));
130 
131  for (uint32_t i = 0; i < slices; ++i)
132  {
133  index().append(c1);
134  index().append(i1 + i);
135  index().append(i1 + i + 1u);
136  }
137  }
138  }
139

146  return std::make_shared<Tube>(name, material, top_radius, base_radius, height, slices);
