55  /*! \brief Constructor with initialization list */
56  Vector(std::initializer_list<T> initList)
57  {
58  const size_t m = std::min(static_cast<size_t>(N), initList.size()); /* FIXME cast */
59  auto iter = initList.begin();
60  for (size_t i = 0_z; i < m; ++i)
61  {
62  m_data[i] = *iter;
63  ++iter;
64  }
65 
66  /* Zero-fill any remaining elements */
67  for (size_t i = m; i < N; ++i)
68  {
69  m_data[i] = maths::zero<T>();
70  }
71  }

153 // *****************************************************************************
154 //! \brief Generic mathematic vector for a given dimension.
155 //!
156 //! \tparam T the type of the vector (float, int).
157 //! \tparam n the vector dimension.
158 //! \note be n small enough to be stored inside the stack of the program without
159 //! making a stack overflow. No checks are made again stack overflow.
160 // *****************************************************************************
161 template <typename T, size_t n>
162 class Vector
163 {
164 public:
165 
166  VECTOR_DIM(n);
167 
168  //! \brief Return the norm of this instance.
169  T norm() const
170  {
171  T l = maths::zero<T>();
172  size_t i = n;
173 
174  while (i--)
175  l += (m_data[i] * m_data[i]);
176  return maths::sqrt(l);
177  }
178 
179  //! \brief Make this instance be a unit vector.
180  //! \note undefined behavior is the norm of the vector is 0.
181  void normalize()
182  {
183  T const l = maths::one<T>() / norm();
184  size_t i = n;
185 
186  while (i--)
187  m_data[i] *= l;
188  }
189 
190  void lerpSelf(Vector<T,n> const& uv, T const alpha)
191  {
192  size_t i = n;
193 
194  while (i--)
195  m_data[i] += (uv.m_data[i] - m_data[i]) * alpha;
196  }
197 
198 protected:
199 
200  //! \brief static array holdind values.
201  //! \note be careful to not produce a stack overflow with a huge size.
202  //! \todo std::array to avoid segfaut in huge dimension ?
203  T m_data[n];
204 };

743  //! \brief Get the coeficient of collinearity (k) of two vectors (u and v).
744  //!
745  //! Two non-null vectors are collinear if and only it exists a
746  //! scalar k != 0, where: u = k v (note: if u is a null vector, any
747  //! vector v is collinear to u because k = 0).
748  //!
749  //! \return k if vectors are collinear, else return NaN if vectors are not
750  //! collinear, else return 0 if zero vector.
751  //!
752  //! \note Use this function for T a familly of float and not for integers.
753  template <typename T, size_t n>
754  T collinearity(Vector<T, n> const& u, Vector<T, n> const& v)
755  {
756  // Null vector ?
757  if (maths::almostZero(u[0]) || maths::almostZero(v[0]))
758  return maths::zero<T>();
759 
760  const T k = u[0] / v[0];
761  for (size_t i = 1_z; i < n; ++i)
762  {
763  if (!maths::almostEqual(k * v[i], u[i]))
764  return T(NAN);
765  }
766  return k;
767  }
