62 static void on_mouse_moved(GLFWwindow* obj, double xpos, double ypos)
63 {
64  static double m_lastX = 0.0;
65  static double m_lastY = 0.0;
66  static bool m_firstMouse = true;
67 
68  assert(nullptr != obj);
69  IGLWindow* window = static_cast<IGLWindow*>(glfwGetWindowUserPointer(obj));
70  window::Mouse& mouse = window->mouse();
71 
72  mouse.position = Vector2g(xpos, ypos);
73 
74  if (m_firstMouse)
75  {
76  m_lastX = xpos;
77  m_lastY = ypos;
78  m_firstMouse = false;
79  }
80 
81  // Reversed since y-coordinates go from bottom to top
82  mouse.displacement = Vector2g(xpos - m_lastX, m_lastY - ypos);
83 
84  m_lastX = xpos;
85  m_lastY = ypos;
86 
87  window->onMouseMoved(mouse);
88 }

148 IGLWindow::IGLWindow(uint32_t const width, uint32_t const height, const char *title)
149  : m_width(width),
150  m_height(height),
151  m_title(title),
152  m_lastKeys(GLFW_KEY_LAST + 1),
153  m_currentKeys(GLFW_KEY_LAST + 1)
154 {
155 
156  DEBUG("%s", "============= OpenGL Context creation ==========================================");
157 
158  if (unlikely(nullptr == m_title)) { m_title = ""; }
159  if (unlikely(0u == m_width)) { m_width = 1u; }
160  if (unlikely(0u == m_height)) { m_height = 1u; }
161 
162  // Initialise GLFW
163  glfwSetErrorCallback(on_GLFW_error);
164  if (!glfwInit())
165  throw OpenGLException("Failed to initialize GLFW");
166 
167  glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE);
168  glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);
169  glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
170  glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
171 
172  // Open a window and create its OpenGL context
173  m_main_window = glfwCreateWindow(static_cast<int>(m_width),
174  static_cast<int>(m_height),
175  m_title, nullptr, nullptr);
176  if (nullptr == m_main_window)
177  throw OpenGLException("Failed to open GLFW window");
178 
179  glfwMakeContextCurrent(m_main_window);
180  glfwSwapInterval(1); // Enable vsync
181 
182  // Initialize GLEW
183  glewExperimental = GL_TRUE; // stops glew crashing on OSX :-/
184  if (GLEW_OK != glewInit())
185  throw OpenGLException("Failed to initialize GLFW");
186 
187  // Print out some info about the graphics drivers
188  std::cout << "OpenGL version: " << glGetString(GL_VERSION) << std::endl;
189  std::cout << "GLSL version: " << glGetString(GL_SHADING_LANGUAGE_VERSION) << std::endl;
190  std::cout << "Vendor: " << glGetString(GL_VENDOR) << std::endl;
191  std::cout << "Renderer: " << glGetString(GL_RENDERER) << std::endl;
192 
193  // Make sure OpenGL version 3.2 API is available
194  if (!GLEW_VERSION_3_2)
195  throw OpenGLException("OpenGL 3.2 API is not available!");
196 
197  setContextCreated();
198 
199  DEBUG("%s", "============= OpenGL Context created ===========================================");
200 }

243 //------------------------------------------------------------------------------
244 void IGLWindow::computeFPS()
245 {
246  static uint32_t nbFrames = 0u;
247  double currentTime = glfwGetTime();
248  m_deltaTime = static_cast<float>(currentTime - m_lastFrameTime);
249  m_lastFrameTime = currentTime;
250  ++nbFrames;
251 
252  if (currentTime - m_lastTime >= 1.0)
253  {
254  // If last prinf() was more than 1sec ago printf and reset
255  std::ostringstream oss;
256  m_fps = nbFrames;
257  int ms_by_frame = static_cast<int>(1000.0f / static_cast<float>(m_fps));
258  oss << '[' << m_fps << " FPS, " << ms_by_frame << " ms] " << m_title;
259  glfwSetWindowTitle(m_main_window, oss.str().c_str());
260  nbFrames = 0u;
261  m_lastTime += 1.0;
262  }
263 }

288 //------------------------------------------------------------------------------
289 bool IGLWindow::start()
290 {
291  // Save the class address to "cast" function callback into a method
292  // callback
293  glfwSetWindowUserPointer(m_main_window, this);
294 
295  // I/O callbacks
296  glfwSetFramebufferSizeCallback(m_main_window, on_window_resized);
297 
298  // Ensure we can capture keyboard being pressed below
299  glfwSetInputMode(m_main_window, GLFW_STICKY_KEYS, GL_TRUE);
300 
301  for (size_t i = GLFW_KEY_SPACE; i <= GLFW_KEY_LAST; ++i)
302  m_lastKeys[i] = m_currentKeys[i] = window::KEY_RELEASE;
303 
304  // Flush OpenGL errors before using this function on real OpenGL
305  // routines else a fake error is returned on the first OpenGL
306  // routines while valid.
307  glCheck();
308 
309  // Init OpenGL states of the user application
310  DEBUG("%s", "============= SETUP ============================================================");
311 
312  try
313  {
314  if (unlikely(!setup()))
315  {
316  ERROR("Failed setting-up graphics");
317  onSetupFailed();
318  return false;
319  }
320 
321  // Force refreshing computation made when window changed
322  onWindowSizeChanged();
323  }
324  catch (const OpenGLException& e)
325  {
326  ERROR("Caught exception during setting-up graphics: '%s'",
327  e.message().c_str());
328  onSetupFailed();
329  return false;
330  }
331 
332  // init FPS
333  m_lastTime = glfwGetTime();
334  m_lastFrameTime = m_lastTime;
335  m_fps = 0;
336 
337  // Draw OpenGL scene
338  return loop();
339 }

341 //------------------------------------------------------------------------------
342 bool IGLWindow::loop()
343 {
344  do
345  {
346  DEBUG("%s", "============= LOOP =============================================================");
347 
348  display_gpu_memory();
349  computeFPS();
350 
351  try
352  {
353  if (likely(false == draw()))
354  {
355  ERROR("Failed drawing graphics");
356  onDrawFailed();
357  return false;
358  }
359  }
360  catch (const OpenGLException& e)
361  {
362  ERROR("Caught exception during drawing graphics: '%s'",
363  e.message().c_str());
364  onDrawFailed();
365  return false;
366  }
367 
368  // Swap buffers
369  glfwSwapBuffers(m_main_window);
370 
371  // Update window events (input etc)
372  {
373  glfwPollEvents();
374  const std::lock_guard<std::mutex> lock(m_mutex);
375  onKeyboardEvent();
376  for (size_t i = GLFW_KEY_SPACE; i <= GLFW_KEY_LAST; ++i)
377  m_lastKeys[i] = m_currentKeys[i];
378  }
379  }
380  // Check if the ESC key was pressed or the window was closed
381  // Note: use keyPressed() not isKeyDown() because if Event::Keyboard is not we
382  // no longer can halt the window.
383  while ((GLFW_PRESS != glfwGetKey(m_main_window, GLFW_KEY_ESCAPE)) &&
384  (0 == glfwWindowShouldClose(m_main_window)));
385  return true;
386 }
