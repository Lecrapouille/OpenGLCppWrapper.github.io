47 // *****************************************************************************
48 //! \brief A GLSampler is an OpenGL uniform for texture.
49 // *****************************************************************************
50 class GLSampler: public GLLocation
51 {
52 public:
53 
54  //----------------------------------------------------------------------------
55  //! \brief See GLLocation constructor.
56  //! \param name
57  //! \param gltype
58  //! \param texture_id count texture.
59  //! \param prog
60  //----------------------------------------------------------------------------
61  GLSampler(const char *name, const GLint gltype, const GLenum texture_id,
62  const GLuint prog)
63  : GLLocation(name, 1, gltype, prog),
64  m_texture_id(texture_id)
65  {
66  redoUpdate();
67  }
68 
69  //----------------------------------------------------------------------------
70  //! \brief Return the texture identifier.
71  //----------------------------------------------------------------------------
72  inline GLint textureID() const
73  {
74  return static_cast<GLint>(m_texture_id);
75  }
76 
77 private:
78 
79  //----------------------------------------------------------------------------
80  //! \brief Create a new OpenGL Uniform.
81  //----------------------------------------------------------------------------
82  virtual bool create() override
83  {
84  DEBUG("Sampler '%s' create", cname());
85  m_handle = glCheck(glGetUniformLocation(m_program, cname()));
86  return false;
87  }
88 
89  //----------------------------------------------------------------------------
90  //! \brief Setup the behavior of the instance. This is a dummy method. No
91  //! action is made.
92  //----------------------------------------------------------------------------
93  virtual bool setup() override
94  {
95  return false;
96  }
97 
98  //----------------------------------------------------------------------------
99  //! \brief
100  //----------------------------------------------------------------------------
101  virtual void activate() override
102  {
103  DEBUG("Sampler '%s' activate GL_TEXTURE0 + %u", cname(), m_texture_id);
104  glCheck(glActiveTexture(GL_TEXTURE0 + m_texture_id));
105  }
106 
107  //----------------------------------------------------------------------------
108  //! \brief
109  //----------------------------------------------------------------------------
110  virtual bool update() override
111  {
112  DEBUG("Sampler '%s' update", cname());
113  glCheck(glUniform1i(m_handle, textureID()));
114  return false;
115  }
116 
117  //----------------------------------------------------------------------------
118  //! \brief Unbind the OpenGL Attribute.
119  //----------------------------------------------------------------------------
120  virtual void deactivate() override
121  {}
122 
123  //----------------------------------------------------------------------------
124  //! \brief Destroy the OpenGL Sampler. This is a dummy method. No action is made.
125  //----------------------------------------------------------------------------
126  virtual void release() override
127  {}
128 
129 protected:
130 
131  const GLenum m_texture_id;
132 };
