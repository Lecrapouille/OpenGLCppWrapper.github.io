146  static inline bool almostEqual(float const A, float const B)
147  {
148  if (A == B)
149  return true;
150 
151  // Make sure maxUlps is non-negative and small enough that the
152  // default NAN won't compare as equal to anything.
153  assert(maths::maxUlps < 4U * 1024U * 1024U);
154 
155  int aInt = *(int*) &A;
156 
157  // Make aInt lexicographically ordered as a twos-complement int
158  if (aInt < 0)
159  aInt = 0x80000000 - aInt;
160 
161  // Make bInt lexicographically ordered as a twos-complement int
162  int bInt = *(int*) &B;
163 
164  if (bInt < 0)
165  bInt = 0x80000000 - bInt;
166 
167  int intDiff = maths::abs(aInt - bInt);
168  if (intDiff <= (int) maths::maxUlps)
169  return true;
170  return false;
171  }

288  //! \brief Return evenly spaced numbers over a specified interval.
289  //! \param[in] start The starting scalar value of the sequence.
290  //! \param[in] end The end value of the sequence, unless endpoint is set to false.
291  //! In that case, the sequence consists of all but the last of num + 1
292  //! evenly spaced samples, so that stop is excluded.
293  //! \param[in] N Number of samples to generate. Should be > 1.
294  //! \param[out] The vector of equally spaced samples in the closed interval [start, stop]
295  //! or the half-open interval [start, stop) (depending on whether endpoint is true or false).
296  //! \return Size of spacing between samples. Return NaN if this value cannot be
297  //! computed.
298  //! \note: This code has been inpsired by the Numpy.linspace function.
299  // *****************************************************************************
300  template<typename T>
301  static T linspace(T const start, T const end, size_t const N, std::vector<T>& result, const bool endpoint)
302  {
303  const T not_a_number = std::numeric_limits<T>::quiet_NaN();
304  result.clear();
305 
306  if (0 == N) return not_a_number;
307  if (1 == N)
308  {
309  result.push_back(start);
310  return not_a_number;
311  }
312 
313  result.reserve(N);
314  T delta = (end - start) / static_cast<T>(N-1);
315  for (size_t i = 0; i < N-1; ++i)
316  {
317  result.push_back(start + delta * T(i));
318  }
319 
320  if (endpoint)
321  {
322  result.push_back(end);
323  }
324  return delta;
325  }
