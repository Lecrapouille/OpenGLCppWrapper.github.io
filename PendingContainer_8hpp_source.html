35 // *****************************************************************************
36 //! \brief PendingContainer is a class over std::vector for memorizing elements
37 //! and knowing which elements have been modified (pending data).
38 //!
39 //! This class is used as buffer between CPU and GPU where pending elements are
40 //! flushed to the GPU memory. Mostly method impacting on the container size
41 //! have a side effect: they modify the estimator on the GPU memory usage for
42 //! the current running application.
43 // *****************************************************************************
44 template<class T>
45 class PendingContainer: public PendingData
46 {
47  // ***************************************************************************
48  //! \brief Allow to PendingContainer to set or get an element in the container
49  //! when using the operator[].
50  //!
51  //! operator[] when not const will always be used as a setter even when it is
52  //! supposed to call a getter. For example a = std::container[0] will call the
53  //! std::container::set() method while desired to call the get() method (in
54  //! the same way than (PendingContainer[0] = a).
55  //!
56  //! For PendingContainer this is not a desirable behavior because this will
57  //! set the nth element as dirty (pending data) and imply a flush to the GPU
58  //! memory in the case of VBO.
59  //!
60  //! To allow C++ to understand to correctly select setter getter to use this
61  //! class is needed.
62  // ***************************************************************************
63  struct Deref
64  {
65  PendingContainer& m_ref;
66  size_t m_index;
67 
68  //! \brief Constructor takes a reference of PendingContainer and the nth
69  //! element of the container.
70  explicit Deref(PendingContainer& container, size_t const nth)
71  : m_ref(container), m_index(nth)
72  {}
73 
74  //! \brief Getter method.
75  inline operator T() const
76  {
77  return m_ref.get(m_index);
78  }
79 
80  //! \brief Setter method.
81  inline T& operator=(T const& other)
82  {
83  return m_ref.set(m_index) = other;
84  }
85 
86  //! \brief This method allow to write std::cout << PendingContainer[0]
87  inline friend std::ostream& operator<<(std::ostream& os, const Deref& d)
88  {
89  return os << d.m_ref.get(d.m_index);
90  }
91  };

227  //----------------------------------------------------------------------------
228  //! \brief Resize the container when this is possible.
229  //! \throw std::out_of_range if the container cannot be resized.
230  //----------------------------------------------------------------------------
231  inline void resize(size_t const count)
232  {
233  size_t old_count = size();
234 
235  // Same size: ignore!
236  if (unlikely((0_z == count) || (count == old_count)))
237  return ;
238 
239  // Resize the container
240  throw_if_cannot_expand();
241  m_container.resize(count);
242  // FIXME not optimized concerning m_pending_start
243  tagAsPending(0_z, m_container.size());
244 
245  // Case of container is reduced: update pending data
246  if (unlikely(count < old_count))
247  {
248  if (0_z == count)
249  {
250  clearPending();
251  }
252  else
253  {
254  size_t pos_start;
255  size_t pos_end;
256  getPendingData(pos_start, pos_end);
257  clearPending();
258  tagAsPending(std::min(pos_start, count), std::min(pos_end, count));
259  }
260  }
261 
262  GPUMemory() += (count * sizeof (T) - memory());
263  DEBUG("'%s': Resizing %zu elements of size %zu bytes. GPU memory: %zu bytes",
264  cdebug(), count, sizeof (T), GPUMemory().load());
265  }
